    def add_subscription(type, endpoint, query, metadata, batchsize, rate, format, addlist='', removelist='')
       body = '{"Subscription" : 
       {"Type" :"' + type + '",
       "Endpoint" : "' + endpoint + '",
       "Query" :{' + query + '} ,
       "MetaData" : "' + metadata + '",
       "BatchSize" : ' + batchsize + ',
       "Rate" :' + rate + ',
       "Format" : "' + format + '",
       "UserList" : {"AddList" : [' + addlist + '],
       "RemoveList" :[' + removelist + ']}}}'
      @context.call_myspace_api(:stream_subscription_post,:body=>body)    
      
    end
