  def set_status(person_id,latitude,longitude,moodName,status)
  @context.validate_identifier(:person_id, person_id)   
  body ='{
            "currentLocation":{
                "latitude":"'+latitude+'",
                "longitude":"'+longitude+'"
            },
            "moodName":"'+moodName+'",
            "status":"'+status+'"
        }'
       @context.call_myspace_api(:status_update_v9,:body=>body)
   end
